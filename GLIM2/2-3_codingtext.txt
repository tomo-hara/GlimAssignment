#include <string>
#include <vector>
#include <iostream>

using namespace std;

struct POINT {
    int x;
    int y;
};

int solution(vector<vector<int>> dots) {
    int answer = 0;
    
    POINT pos[4];
    for (int i=0; i<dots.size(); i++) {
        pos[i].x = dots[i][0];
        pos[i].y = dots[i][1];
    }
    int width = abs(pos[0].x - pos[1].x) | abs(pos[0].x - pos[2].x);
    int height = abs(pos[0].y - pos[1].y) | abs(pos[0].y - pos[2].y);
    answer = width * height;
    
    return answer;
}
